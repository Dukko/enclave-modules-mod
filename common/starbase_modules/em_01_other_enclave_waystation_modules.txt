# =============================================================================
# ENCLAVE MODULES MOD - Other Enclave Waystation Modules
# Nomadic equivalents of the Curator Archive, Artisan Stage, Numistic Exchange.
#
# Design philosophy (mirrors the Shroud/Warden nomadic modules):
#   - Fixed buildings anchor and deepen. Waystation modules are mobile, networked,
#     and share their benefits across the wayline rather than concentrating them.
#   - Bonuses are slightly lighter per-station but benefit the whole network.
#
# FLAG NOTES (verified against vanilla files):
#   curator_enclave_country   - country flag on the Curator enclave            ✓
#   artist_enclave_country    - country flag on the Artisan Troubadour enclave  ✓ (NOT artisan_)
#   caravaneer_home_country   - country flag on the Caravaneer home station     ✓
# =============================================================================

# =============================================================================
# CURATOR SURVEY TERMINAL
# A Curator field office for wandering minds. The Curators have found nomadic
# empires make ideal survey partners — their fleets visit systems no fixed
# archive ever reaches, accumulating observations the Order would pay dearly for.
# The Terminal formalizes this exchange: your sensor data for their analytical
# frameworks. Insights propagate across the wayline.
# =============================================================================
waystation_curator_survey_terminal = {
	icon = "GFX_em_curator_icon"
	section = "SCIENCE_STARBASE_SECTION"
	construction_days = 180
	starbase_type = waystation
	category = sm_enclave_cat

	starbase_limit = {
		base = 1
	}

	potential = {
		exists = owner
		is_waystation_starbase = yes
		owner = {
			is_homicidal = no
			has_country_flag = em_curator_partnership
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "em_already_built"
			owner = {
				NOT = {
					any_owned_starbase = {
						NOT = { is_same_value = root }
						has_starbase_module = waystation_curator_survey_terminal
					}
				}
			}
		}
	}

	resources = {
		category = starbase_waystations_modules
		cost = {
			alloys = 100
		}
		upkeep = {
			trade = 2
		}
	}

	# Base: Curator analysts review local sensor data at each system visited
	station_modifier = {
		starbase_collected_system_physics_research_add = 3
		starbase_collected_system_society_research_add = 3
	}

	# Anomaly generation improves empire-wide while the Terminal is active
	empire_modifier = {
		ship_anomaly_generation_chance_mult = 0.05
	}

	# Gestalt bonus: the Curators find collective processing uniquely valuable
	triggered_station_modifier = {
		potential = {
			owner? = { is_gestalt = yes }
		}
		starbase_collected_system_physics_research_add = 2
		starbase_collected_system_engineering_research_add = 2
	}

	# Network: Curator analytical frameworks propagate across the wayline
	waystation_network_country_modifier = {
		custom_tooltip = wayline_enclave_effects_tt
		show_only_custom_tooltip = no
		country_physics_tech_research_speed = 0.03
		country_society_tech_research_speed = 0.03
	}

	# Joyous Enlightenment: a network of surveying minds amplifies every discovery
	triggered_station_modifier = {
		potential = {
			owner? = { has_country_flag = sacred_path_joyous_enlightenment }
			has_neighboring_waystation = yes
		}
		custom_tooltip = blessings_joyous_enlightenment_tt
		show_only_custom_tooltip = no
		starbase_collected_system_physics_research_add = 3
		starbase_collected_system_society_research_add = 3
	}

	# Middle Path - not a sacred site
	triggered_station_modifier = {
		potential = {
			owner? = { has_country_flag = sacred_path_middle_path }
			is_sacred_site_waystation = no
		}
		custom_tooltip = bonus_middle_path_tt
		show_only_custom_tooltip = no
		starbase_collected_system_physics_research_add = 2
	}

	# Middle Path - sacred site: ancient places hold data the Curators hunger for
	triggered_station_modifier = {
		potential = {
			owner? = { has_country_flag = sacred_path_middle_path }
			is_sacred_site_waystation = yes
		}
		custom_tooltip = bonus_sacred_site_tt
		show_only_custom_tooltip = no
		starbase_collected_system_physics_research_add = 3
		starbase_collected_system_society_research_add = 3
	}

	ai_weight = {
		weight = 100
		modifier = {
			factor = 2
			owner? = {
				OR = {
					has_ethic = ethic_materialist
					has_ethic = ethic_fanatic_materialist
					is_gestalt = yes
				}
			}
		}
	}
}
