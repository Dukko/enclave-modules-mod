# =============================================================================
# ENCLAVE MODULES MOD - Starbase Buildings
# New buildings for Shroudwalker and Mindwarden enclaves.
# For traditional (non-nomadic) empires only.
#
# FLAG NAME NOTES (verify against vanilla Overlord DLC files if needed):
#   shroudwalker_enclave_country        - country flag set on the Shroudwalker enclave's own country
#   shroudwalker_enclave_diplomacy_engaged - set on player country at first Shroudwalker contact
#   is_mercenary_mindwarden_enclave     - scripted trigger for Mindwarden enclave country
#   (Mindwardens have no "diplomacy engaged" flag; identified via scripted trigger only)
# =============================================================================

# =============================================================================
# SHROUD CONDUIT
# A permanent conduit to the Shroud, anchored in collaboration with the
# Shroudwalker Enclave. Unlike the nomadic Communion Chamber, this installation
# builds up a deep, resonant connection over time in one place.
#
# Psionic empires commune directly, gaining spiritual insight and Zro.
# Non-psionic empires study the phenomenon academically, gaining research.
# Spiritualist empires find their citizens moved by proximity to the Shroud.
# =============================================================================
em_shroud_conduit = {
	icon = "GFX_em_shroudwalker_icon"
	construction_days = 360

	potential = {
		exists = owner
		is_normal_starbase = yes
		owner = {
			is_homicidal = no
			# Set on the player's country when they first engage with the Shroudwalker Enclave
			has_country_flag = shroudwalker_enclave_diplomacy_engaged
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "em_already_built"
			owner = {
				NOT = {
					any_owned_starbase = {
						NOT = { is_same_value = root }
						has_starbase_building = em_shroud_conduit
					}
				}
			}
		}
	}

	resources = {
		category = starbase_buildings
		cost = {
			energy = 750
		}
		upkeep = {
			energy = 3
		}
		# Base: unity for all empires who commune here
		produces = {
			unity = 5
		}
		# Psionic empires connect more deeply — stronger unity surge
		produces = {
			trigger = {
				exists = owner
				owner = {
					OR = {
						has_tradition = tr_psionics_adopt
						species = { has_trait = trait_psionic }
					}
				}
			}
			unity = 5
		}
		# Psionic empires receive direct Shroud-sourced insights
		produces = {
			trigger = {
				exists = owner
				owner = {
					OR = {
						has_tradition = tr_psionics_adopt
						species = { has_trait = trait_psionic }
					}
				}
			}
			society_research = 5
		}
		# Non-psionic empires study the Shroud as phenomenon, not experience
		produces = {
			trigger = {
				exists = owner
				owner = {
					NOR = {
						has_tradition = tr_psionics_adopt
						species = { has_trait = trait_psionic }
						species = { has_trait = trait_latent_psionic }
					}
				}
			}
			physics_research = 3
			society_research = 2
		}
		# Spiritualist empires: citizens are moved by the Conduit's presence
		produces = {
			trigger = {
				exists = owner
				owner = {
					OR = {
						has_ethic = ethic_spiritualist
						has_ethic = ethic_fanatic_spiritualist
					}
				}
			}
			unity = 3
		}
		# Psionic empires with Zro mining tap into residual Shroud energy
		produces = {
			trigger = {
				exists = owner
				owner = {
					has_technology = tech_mine_zro
					OR = {
						has_tradition = tr_psionics_adopt
						species = { has_trait = trait_psionic }
					}
				}
			}
			sr_zro = 1
		}
	}

	# Empire-wide: the Shroud's wisdom eases the path of tradition
	country_modifier = {
		tradition_cost_mult = -0.05
	}

	# Spiritualist bonus: the Conduit galvanizes faith empire-wide
	triggered_country_modifier = {
		potential = {
			owner = {
				OR = {
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
				}
			}
		}
		tradition_cost_mult = -0.03
		unity_from_jobs_mult = 0.05
	}

	# Psionic ascension bonus: the Conduit amplifies active shroud contact
	triggered_country_modifier = {
		potential = {
			owner = {
				OR = {
					has_tradition = tr_psionics_adopt
					species = { has_trait = trait_psionic }
				}
			}
		}
		shroud_passive_psionic_mult = 0.10
	}

	ai_weight = {
		weight = 100
		modifier = {
			factor = 3
			owner = {
				OR = {
					has_tradition = tr_psionics_adopt
					species = { has_trait = trait_psionic }
				}
			}
		}
		modifier = {
			factor = 2
			owner = {
				OR = {
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
				}
			}
		}
		modifier = {
			factor = 0
			owner = { is_nomadic = yes }
		}
	}
}

# =============================================================================
# WARDEN'S VIGIL
# A permanent monitoring installation staffed with Mindwarden expertise.
# Watches for psionic anomalies, disrupts unauthorized Shroud contact, and
# anchors Mindwarden logistical networks in the region.
#
# Unlike the nomadic Monitoring Post, the Vigil is fortified and permanent —
# a deterrent as much as an intelligence tool.
#
# Militarist empires channel Warden discipline into ground forces.
# Academic/materialist empires study the suppression technology itself.
# Machine intelligences make exceptional partners — no psionic vulnerabilities.
#
# Cannot be built by empires with psionic founder species.
# =============================================================================
em_warden_vigil = {
	icon = "GFX_em_mindwarden_icon"
	construction_days = 360

	potential = {
		exists = owner
		is_normal_starbase = yes
		owner = {
			is_homicidal = no
			has_country_flag = em_mindwarden_partnership
			# Psionic empires are rejected by the Mindwardens entirely
			species = {
				NOT = { has_trait = trait_psionic }
				NOT = { has_trait = trait_latent_psionic }
			}
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "em_already_built"
			owner = {
				NOT = {
					any_owned_starbase = {
						NOT = { is_same_value = root }
						has_starbase_building = em_warden_vigil
					}
				}
			}
		}
	}

	resources = {
		category = starbase_buildings
		cost = {
			energy = 500
			alloys = 100
		}
		upkeep = {
			energy = 3
		}
		# Non-militarist empires: research into psionic suppression technology
		produces = {
			trigger = {
				exists = owner
				owner = {
					NOR = {
						has_ethic = ethic_militarist
						has_ethic = ethic_fanatic_militarist
					}
				}
			}
			physics_research = 5
		}
	}

	# Core: psionic detection and structural hardening
	station_modifier = {
		ship_cloaking_detection_add = 2
		shipclass_starbase_hull_mult = 0.05
	}

	# Mindwarden logistical support: supply chain coordination
	country_modifier = {
		ship_upkeep_mult = -0.05
	}

	# Militarist empires: Warden combat doctrine applied to planetary assault forces
	triggered_country_modifier = {
		potential = {
			owner = {
				OR = {
					has_ethic = ethic_militarist
					has_ethic = ethic_fanatic_militarist
				}
			}
		}
		country_army_damage_mult = 0.10
		army_defense_morale_mult = 0.10
	}

	# Machine intelligence: perfect anti-psionic partners, zero susceptibility
	triggered_station_modifier = {
		potential = {
			owner = {
				is_machine_empire = yes
			}
		}
		ship_cloaking_detection_add = 1
	}

	# Machine empire country bonus: the Mindwardens trust machine logic
	triggered_country_modifier = {
		potential = {
			owner = {
				is_machine_empire = yes
			}
		}
		ship_upkeep_mult = -0.05
	}

	ai_weight = {
		weight = 100
		modifier = {
			factor = 2
			owner = {
				OR = {
					has_ethic = ethic_militarist
					has_ethic = ethic_fanatic_militarist
				}
			}
		}
		modifier = {
			factor = 2
			owner = { is_machine_empire = yes }
		}
		modifier = {
			factor = 0
			owner = {
				OR = {
					is_nomadic = yes
					species = { has_trait = trait_psionic }
					species = { has_trait = trait_latent_psionic }
				}
			}
		}
	}
}
